    public partial class Form2 : Form
    {
        string connectionString = "server=localhost;database=usknigga1;uid=root;pwd=;";
        MySqlConnection connection;
        DataTable ListTable;
        string currentUsername = "";
        string currentRole = "";
        int currentId = 0;
        bool isEditing = false;

        public Form2(string username, string fullname, string role)
        {
            InitializeComponent();
            currentUsername = username;
            currentRole = role;
            InitializeForm(fullname);
            CheckRole();
        }

        private void InitializeForm(string fullname)
        {
            connection = new MySqlConnection(connectionString);
            lblFn.Text = "halo ! " + fullname;

            GridView.SelectionChanged += GridView_SelectionChanged;
            GridView.CellFormatting += GridView_CellFormatting;
            GridView.CellContentClick += GridView_CellContentClick;
        }

        private void Form2_Load(object sender, EventArgs e)
        {
            LoadData();
            ClearFields();
            ButtonVisibility(false);
        }

        private void LoadData()
        {
            connection.Open();
            string query = "SELECT id, title, description, finished_at FROM list WHERE username = @username ORDER BY created_at DESC";
            MySqlCommand cmd = new MySqlCommand(query, connection);
            cmd.Parameters.AddWithValue("@username", currentUsername);

            MySqlDataAdapter adapter = new MySqlDataAdapter(cmd);
            ListTable = new DataTable();
            adapter.Fill(ListTable);
            SetupData();
            connection.Close();
        }

        private void SetupData()
        {
            GridView.DataSource = null;
            GridView.Columns.Clear();
            GridView.AllowUserToAddRows = false;

            DataGridViewCheckBoxColumn checkbox = new DataGridViewCheckBoxColumn();
            checkbox.Name = "Status";
            checkbox.ReadOnly = false;
            GridView.Columns.Add(checkbox);

            GridView.DataSource = ListTable;

            if (GridView.Columns["id"] != null)
            {
                GridView.Columns["id"].Visible = false;
            }
        }
        private void ExecuteQuery(string query, params MySqlParameter[] parameters)
        {
            connection.Open();
            MySqlCommand cmd = new MySqlCommand(query, connection);
            if (parameters != null) cmd.Parameters.AddRange(parameters);
            cmd.ExecuteNonQuery();
            connection.Close();
        }

        private void ButtonVisibility(bool visible)
        {
            btnEdit.Visible = visible;
            btnDelete.Visible = visible;
        }

        private void ClearFields()
        {
            txtBoxTitle.Text = "";
            txtBoxDesc.Text = "";
            isEditing = false;
            currentId = 0;
            GridView.ClearSelection();
        }

        private void GridView_SelectionChanged(object? sender, EventArgs e)
        {
            if (GridView.CurrentCell != null && GridView.CurrentCell.RowIndex >= 0)
            {
                ButtonVisibility(true);
            }
            else
            {
                ButtonVisibility(false);
            }
        }

        private void CheckRole()
        {
            if (currentRole == "admin")
            {
                btnUser.Visible = true; 
            }
            else
            {
                btnUser.Visible = false; 
            }
        }

        private void GridView_CellFormatting(object? sender, DataGridViewCellFormattingEventArgs e)
        {
            if (e.RowIndex >= 0 && e.RowIndex < GridView.Rows.Count)
            {
                DataGridViewRow gridRow = GridView.Rows[e.RowIndex];
                DataRowView dataRow = gridRow.DataBoundItem as DataRowView;

                if (dataRow != null)
                {
                    if (GridView.Columns[e.ColumnIndex].Name == "Status")
                    {
                        bool isFinished = dataRow["finished_at"] != DBNull.Value;
                        e.Value = isFinished;
                    }
                    if (dataRow["finished_at"] != DBNull.Value)
                    {
                        gridRow.DefaultCellStyle.BackColor = Color.LightGreen;
                    }
                    else
                    {
                        gridRow.DefaultCellStyle.BackColor = Color.White;
                    }
                }
            }
        }

        private void GridView_CellContentClick(object? sender, DataGridViewCellEventArgs e)
        {
            if (e.RowIndex >= 0 && GridView.Columns[e.ColumnIndex].Name == "Status")
            {
                GridView.CommitEdit(DataGridViewDataErrorContexts.Commit);
                DataRowView row = (DataRowView)GridView.Rows[e.RowIndex].DataBoundItem;
                int id = Convert.ToInt32(row["id"]);
                bool isChecked = (bool)GridView.Rows[e.RowIndex].Cells["Status"].Value;

                if (isChecked)
                {
                    SoundPlayer audio = new SoundPlayer(@"C:\Users\g\source\repos\usksksksk\usksksksk\bin\Debug\boom.wav");
                    audio.Play();
                }

                string query = isChecked ? "UPDATE list SET finished_at = NOW() WHERE id = @id" : "UPDATE list SET finished_at = NULL WHERE id = @id";

                ExecuteQuery(query, new MySqlParameter("@id", id));
                LoadData();
            }

        }

        private void btnSave_Click(object sender, EventArgs e)
        {
            if (string.IsNullOrEmpty(txtBoxTitle.Text))
            {
                MessageBox.Show("Title must be filled!!", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                txtBoxTitle.Focus();
                return;
            }

            string query = isEditing
                ? "UPDATE list SET title = @title , description = @desc WHERE id = @id AND username = @username"
                : "INSERT into list (username, title, description) VALUES (@username, @title, @desc)";

            MySqlCommand cmd = new MySqlCommand(query, connection);
            cmd.Parameters.AddWithValue("@username", currentUsername);
            cmd.Parameters.AddWithValue("@title", txtBoxTitle.Text);
            cmd.Parameters.AddWithValue("@desc", txtBoxDesc.Text);
            if (isEditing) cmd.Parameters.AddWithValue("@id", currentId);

            connection.Open();
            cmd.ExecuteNonQuery();
            connection.Close();
            ClearFields();
            LoadData();
        }

        private void btnEdit_Click(object sender, EventArgs e)
        {
            if (GridView.CurrentRow != null)
            {
                DataRowView rowView = GridView.CurrentRow.DataBoundItem as DataRowView;
                if (rowView != null)
                {
                    currentId = Convert.ToInt32(rowView["id"]);
                    txtBoxTitle.Text = rowView["title"].ToString();
                    txtBoxDesc.Text = rowView["description"].ToString();
                    isEditing = true;
                }
            }
        }

        private void btnDelete_Click(object sender, EventArgs e)
        {
            if (GridView.CurrentRow != null)
            {
                int rowindex = GridView.CurrentCell.RowIndex;
                int id = Convert.ToInt32(ListTable.Rows[rowindex]["id"]);

                DialogResult dialogresult = MessageBox.Show("Sure want to delete?", "Confirm", MessageBoxButtons.YesNo);
                if (dialogresult == DialogResult.Yes)
                {
                    string query = "DELETE FROM list where id = @id";
                    ExecuteQuery(query, new MySqlParameter("@id", id));
                    ClearFields();
                    LoadData();
                }
            }
        }
        private void txtBoxSearch_TextChanged(object sender, EventArgs e)
        {
            if (GridView != null)
            {
                string keyword = txtBoxSearch.Text.Replace("'", "''");
                ListTable.DefaultView.RowFilter = string.Format("title like '%{0}%'", keyword);

                ClearFields();
            }
        }

        private void btnLogout_Click(object sender, EventArgs e)
        {
            DialogResult dialogresult = MessageBox.Show("are you sure?", "confirm", MessageBoxButtons.YesNo, MessageBoxIcon.Question);
            if (dialogresult == DialogResult.Yes)
            {
                Form1 gate = new Form1();
                gate.Show();
                this.Dispose();
            }
        }
    }
