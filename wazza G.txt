using MySql.Data.MySqlClient;

namespace usksksksk
{
    public partial class Form1 : Form
    {
        string connectionString = "Server=localhost;database=usknigga1;uid=root;pwd=;";
        MySqlConnection connection;
        bool typeusn = false;
        bool typepw = false;

        public Form1()
        {
            InitializeComponent();
            InitializeForm();
        }

        private void InitializeForm()
        {
            connection = new MySqlConnection(connectionString);
            lblWarnUsn.Visible = false;
            lblWarnPw.Visible = false;
            txtBoxPw.UseSystemPasswordChar = true;
        }

        private void txtBoxUsn_TextChanged(object sender, EventArgs e)
        {
            if (txtBoxUsn.Text.Length > 0)
            {
                typeusn = true;
                lblWarnUsn.Visible = false;
            }
            if (typeusn && txtBoxUsn.Text.Length == 0)
            {
                lblWarnUsn.Visible = true;
            }
        }

        private void txtBoxPw_TextChanged(object sender, EventArgs e)
        {
            if (txtBoxPw.Text.Length > 0)
            {
                typepw = true;
                lblWarnPw.Visible = false;
            }
            if (typepw && txtBoxPw.Text.Length == 0)
            {
                lblWarnPw.Visible = true;
            }
        }

        private void checkBoxPw_CheckedChanged(object sender, EventArgs e)
        {
            txtBoxPw.UseSystemPasswordChar = !checkBoxPw.Checked;
        }

        private void btnLogin_Click(object sender, EventArgs e)
        {
            string username = txtBoxUsn.Text.Trim();
            string password = txtBoxPw.Text.Trim();

            if (string.IsNullOrEmpty(username))
            {
                MessageBox.Show("Username harus dimasukkan!", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                txtBoxUsn.Focus();
                return;
            }
            if (string.IsNullOrEmpty(password))
            {
                MessageBox.Show("Password harus dimasukkan!", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                txtBoxUsn.Focus();
                return;
            }

            Auth(username,password);
        }

        private void Auth(string username, string password)
        {
            try
            {
                connection.Open();
                string query = "SELECT fullname FROM users WHERE username = @user and password = @pass";
                MySqlCommand cmd = new MySqlCommand(query, connection);
                cmd.Parameters.AddWithValue("@user", username);
                cmd.Parameters.AddWithValue("@pass", password);
                object result = cmd.ExecuteScalar();
                connection.Close();

                if(result != null)
                {
                    string fullname = result.ToString();
                    MessageBox.Show("Selamat datang " + fullname + "!!");

                    Form2 list = new Form2(fullname);
                    list.Show();
                    this.Hide();
                }
                else
                {
                    MessageBox.Show("Username / Password salah!!", "Error!!", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            }
            catch(Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
        }
    }
}
